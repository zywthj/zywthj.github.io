常用的封装函数

  • 封装函数

自己积累的一些常用小函数，喜欢的可以瞅瞅哦~

1、向顶部滚动

a为锚点，speed为向上滚动速度，有三个值：normal, slow, fast

function backTop(a, speed){
    window.onscroll = function(){
        var backtop = $("body").scrollTop();
        var oPage = $(window).innerHeight();
        if(backtop > oPage){
            $(a).css("display", "block");
        } else {
            $(a).css("display", "none");
        }
    };
    $(a).click(function(){
        $("html, body").animate({scrollTop : 0}, speed);
        return false;
    });
}

2、鼠标放上时逐渐增强或减弱的透明效果

element为选中元素，speed为透明度切换速度，有三个值：normal, slow, fast

function opacityChange(element, speed){
    $(element).fadeTo(speed, 0.7); 
    $(element).hover(function(){
        $(this).stop(true,true);  //防止鼠标快速放上离开导致多次重复执行动画
        $(this).fadeTo(speed, 1.0); 
    },function(){
        $(this).stop(true,true);
        $(this).fadeTo(speed, 0.7); 
    });
}

3、外部链接添加target=”_blank”属性

a为需要设置的链接

function targetBlank(a){
    var root = location.protocol + '//' + location.host; 
    //location.protocol为协议(常见http/https), location.host为域名和端口
    $(a).not(':contains(root)').click(function(){
        this.target = "_blank";
    });
}

4、清除字符串前后的空格(兼容所有浏览器)

function trim(str) {
    if (str && typeof str === "string") {
        return str.replace(/(^\s*)|(\s*)$/g,""); //去除前后空白符
    }
}

5、添加事件

function addLoadEvent(fn){
    var load = window.onload;  //将全局加载存到一个变量中
    if(typeof window.onload != 'function'){
        window.onload = fn;
    }else{
        window.onload = function(){
            load();  //加载全部传入的函数，如果未定义全局加载则只能加载一个。
            fn();
        }
    }
}

6、将某元素插入目标元素后面

newElement为新元素，targetElement为目标元素

function insertAfter(newElement, targetElement){
    var parent = targetElement.parentNode;
    if(parent.lastChild == targetElement){
        parent.appendChild(newElement);
    }else{
        parent.insertBefore(newElement, targetElement.nextSibling);
    }
}

7、js获取地址栏参数

name为地址栏参数名，如get.html?id=12&url=http://www.baidu.com, name为id或url

function getQueryString(name){
    var reg = new RegExp("(^|&)"+ name +"=([^&]*)(&|$)");
    var r = window.location.search.substr(1).match(reg);
    if(r!=null){
        return  unescape(r[2]);
    } else {
        return null;
    }       
}

8、ajax

function ajax(method, url, data, success) {
    var xhr = null;
    try {
        xhr = new XMLHttpRequest();
    } catch (e) {
        xhr = new ActiveXObject('Microsoft.XMLHTTP');
    }
    if (method == 'get' && data) {
        url += '?' + data;
    }   
    xhr.open(method,url,true);
    if (method == 'get') {
        xhr.send();
    } else {
        xhr.setRequestHeader('content-type', 'application/x-www-form-urlencoded');
        xhr.send(data);
    }

    xhr.onreadystatechange = function() {
        if ( xhr.readyState == 4 ) {
            if ( xhr.status == 200 ) {
                success && success(xhr.responseText);
            } else {
                alert('出错了,Err：' + xhr.status);
            }
        }   
    }
}
